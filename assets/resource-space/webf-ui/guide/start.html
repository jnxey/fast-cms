快速入手 #

通过npm安装 #

在项目中使用 NrWebfUI 时，可通过 npm 安装：

shell
npm install nr_webf_ui
1

你也可以使用 yarn 或 pnpm 进行安装：

shell
# 通过 yarn 安装
yarn add nr_webf_ui

# 通过 pnpm 安装
pnpm add nr_webf_ui

组件使用 #

1.main.ts引入

typescript
import { createApp } from 'vue';
import App from './App.vue';
import { router } from './router';
import NrWebFUI from 'nr_webf_ui';

const app = createApp(App);

app.use(router);
app.use(NrWebFUI);

app.mount(document.body);

2.router.ts使用组件库内置路由方式

typescript
import { createRouter, type RouterHistory } from 'vue-router';
import NrWebFUI from '@/packages';
import IndexView from '@/example/views/IndexView.vue';

const history: RouterHistory = NrWebFUI.createNrMemoryHistory(location.pathname);

export const router = createRouter({
    history: history,
    routes: [
        { path: '/index', name: 'list', component: IndexView },
    ]
});

4.使用函数式组件

vue
<script setup lang="ts">
import NrButton from 'nr_webf_ui/packages/button';
import { useAlert } from 'nr_webf_ui/packages/alert';
const open = (): void => {
    useAlert({
        title: '测试标题',
        content: '测试内容测试内容测试内容测试内容测试内容测试内容'
    })
        .then(() => {
            console.log('点击了确认');
        })
        .catch(() => {
            console.log('直接关闭了');
        });
};
</script>
<template>
    <div class="p-32">
        <!--        -->
        <div class="mb-48">
            <div class="p-20">弹窗</div>
            <NrButton @click="open">点击弹出</NrButton>
        </div>
    </div>
</template>
