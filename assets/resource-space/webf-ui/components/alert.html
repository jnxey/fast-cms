NrAlert 警告框 #

介绍 #

用于中断用户操作，直到用户确认后才可关闭

基础用法 #

vue
<script setup lang="ts">
import NrButton from 'nr_webf_ui/packages/button';
import { useAlert } from 'nr_webf_ui/packages/alert';

const open = (): void => {
    useAlert({
        title: '测试标题',
        content: '测试内容测试内容测试内容测试内容测试内容测试内容'
    })
        .then(() => {
            console.log('点击了确认');
        })
        .catch(() => {
            console.log('直接关闭了');
        });
};
</script>
<template>
    <div class="p-32">
        <!--        -->
        <div class="mb-48">
            <div class="p-20">弹窗</div>
            <NrButton @click="open">点击弹出</NrButton>
        </div>
    </div>
</template>
