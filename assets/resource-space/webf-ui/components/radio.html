NrRadio 单选框 #

介绍 #

用于在一组备选项中进行单选

基础用法 #

vue
<script setup lang="ts">
import { NrRadio, NrRadioGroup } from 'nr_webf_ui/packages/radio';
import NrButton from 'nr_webf_ui/packages/button';
import type { Ref } from 'vue';
import { ref } from 'vue';

const model1: Ref<string | null> = ref(null);

const model2: Ref<string | null> = ref(null);

const list: Ref<string[]> = ref(['1', '2', '3']);
const add = () => {
    list.value = ['1', '2', '3'];
};
const remove = () => {
    list.value = ['1', '3'];
};
</script>
<template>
    <div class="p-32">
        <!--        -->
        <div class="mb-48">
            <div class="p-20">单选框</div>
            <NrRadioGroup class="mb-48" v-model="model1">
                <NrRadio class="mb-20" label="1"> 选项1 </NrRadio>
                <NrRadio class="mb-20" label="2" disabled> 选项2 禁用 </NrRadio>
                <NrRadio class="mb-20" label="3"> 选项3 </NrRadio>
            </NrRadioGroup>
        </div>
        <!--        -->
        <div class="mb-48">
            <div class="p-20">单选框 - 动态</div>
            <NrRadioGroup class="mb-48" v-model="model2">
                <NrRadio class="mb-20" v-for="item in list" :key="item" :label="item"> 选项{{ item }} </NrRadio>
            </NrRadioGroup>
            <NrButton class="mb-20" @click="remove" type="plain">减2</NrButton>
            <NrButton class="mb-20" @click="add" type="plain">加2</NrButton>
        </div>
    </div>
</template>
